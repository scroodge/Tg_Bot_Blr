# Пример .env файла
# Скопируйте в .env и заполните своими данными

# Telegram Bot Token
TELEGRAM_BOT_TOKEN=your_bot_token_here

# Admin User IDs (через запятую)
ADMIN_USER_IDS=123456789,987654321,555666777

# Gemini API Key (для использования с параметром -google)
# Получите ключ в Google AI Studio: https://aistudio.google.com/
GEMINI_API_KEY=your_gemini_api_key_here

# DeepSeek API Key (для использования с параметром --deepseek)
# Получите ключ в DeepSeek Platform: https://platform.deepseek.com/
DEEPSEEK_API_KEY=your_deepseek_api_key_here
